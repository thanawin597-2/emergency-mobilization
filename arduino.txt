#include <WiFi.h>
#include <PubSubClient.h>
#include <ArduinoJson.h>

const char* ssid = "iPhone_aue";
const char* password = "thanawin29";

const char* mqtt_server = "172.20.10.8";

WiFiClient espClient;
PubSubClient client(espClient);

#define BUTTON_PIN 18
#define RELAY_PIN 23

const int OFFICER_ID = 1;
const char* DEVICE_ID = "house_01";

bool lastButtonState = HIGH;
bool mobilized = false;
bool alarmActive = false;
unsigned long lastHeartbeat = 0;

void setup() {
  client.setCallback(callback);

  Serial.begin(115200);

  pinMode(BUTTON_PIN, INPUT_PULLUP);

  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, HIGH);

  WiFi.begin(ssid, password);

  Serial.println("Connecting WiFi...");

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("");
  Serial.println("WiFi Connected");

  Serial.print("ESP32 IP: ");
  Serial.println(WiFi.localIP());

  client.setServer(mqtt_server, 1883);
}

void reconnect() {

  alarmActive = false;

  while (!client.connected()) {

    Serial.println("Connecting MQTT...");

   if (client.connect(DEVICE_ID)) {

    Serial.println("MQTT Connected");

    client.subscribe("ems/mobilize");

    Serial.println("Subscribed: ems/mobilize");

  } else {

      Serial.print("Failed: ");
      Serial.println(client.state());

      delay(3000);
    }
  }
}

void loop() {

  if (!client.connected()) {
    reconnect();
  }

  client.loop();

  if (alarmActive) {

  digitalWrite(
    RELAY_PIN,
    LOW
  );

} else {

  digitalWrite(
    RELAY_PIN,
    HIGH
  );

}

  bool buttonState =
    digitalRead(BUTTON_PIN);

  if (
  mobilized &&
  lastButtonState == HIGH &&
  buttonState == LOW
    ){

    String payload =
      "{\"id\":" +
      String(OFFICER_ID) +
      ",\"status\":\"ack\"}";

    client.publish(
      "ems/ack/house_01",
      payload.c_str()
    );

    Serial.println("ACK SENT");

    alarmActive = false;
    mobilized = false;

    delay(300);
  }

  if (alarmActive) {

  static unsigned long lastPrint = 0;

  if (
    millis() - lastPrint > 1000
  ) {

    Serial.println(
      "ALARM ACTIVE"
    );

    lastPrint = millis();
  }

}

if (
  millis() - lastHeartbeat > 5000
)
{
  String heartbeat =
  "{\"id\":" +
  String(OFFICER_ID) +
  ",\"status\":\"online\"}";

client.publish(
  "ems/status",
  heartbeat.c_str()
);

  Serial.println(
    "HEARTBEAT SENT"
  );

  lastHeartbeat = millis();
}

  lastButtonState = buttonState;
}
void callback(
  char* topic,
  byte* payload,
  unsigned int length
) {

  String msg = "";

  for(int i=0;i<length;i++){
    msg += (char)payload[i];
  }

  Serial.print("TOPIC: ");
  Serial.println(topic);

  Serial.print("MESSAGE: ");
  Serial.println(msg);

  if (String(topic) == "ems/mobilize") {

  DynamicJsonDocument doc(256);

  DeserializationError error =
    deserializeJson(doc, msg);

  if (error) {

    Serial.println("JSON Parse Error");
    return;
  }

  String event =
    doc["event"];

  if (event == "mobilize") {

  Serial.println(
    "===== MOBILIZE RECEIVED ====="
  );

  mobilized = true;
  alarmActive = true;

  Serial.println(
    "Waiting for button..."
  );

}

}

}
